def self.authenticate(kiosk_id, gsw_pin)
  kiosk = Kiosk.where(:kiosk_name => kiosk_id).first
  user = User.where(:gsw_id => gsw_pin).first
  if kiosk && user
    kiosk
  else
    nil
  end
end
