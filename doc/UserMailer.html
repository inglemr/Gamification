def organization_status_change(user,organization)
  @user = user
  @organization = organization

  if(@organization.active)
    @status = "Activated"
  else
    @status = "Deactivated"
  end

  mail(to: @user.email, subject: "[Gamification] Organization #{@organization.name} has been #{@status}")
end
